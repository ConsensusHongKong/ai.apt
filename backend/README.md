## Getting Started


```bash
python3 -m venv aptenv

source ./aptenv/bin/activate

pip install -r requirements.txt

export CRIPTO_API_KEY=<YOUR_CRIPTO_API_KEY>
export API_KEY=<YOUR_DEEPSEEK_API_KEY>
export BASE_URL=<YOUR_DEEPSEEK_API_BASE_URL>
export X_API_KEY=<YOUR_X_API_KEY>
export X_BASE_URL=<YOUR_X_BASE_URL>
export TSM_API_KEY=<YOUR_TSM_API_KEY>
export TSM_API_URL=<YOUR_TSM_API_URL>
export REDIRECT_URI=<YOUR_PROJECT_URI/callback>

python3 api.py

```
